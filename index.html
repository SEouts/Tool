function markEmployeeWithStatus(status) { const nameInput = document.getElementById('manualEmployeeName'); const employeeName = nameInput.value.trim(); if (!employeeName) { showMessage('Please enter an employee name', 'error'); return; } const employeeIndex = workingEmployees.findIndex(emp => emp.name.toLowerCase().includes(employeeName.toLowerCase()) ); if (employeeIndex === -1) { showMessage('Employee not found in working list. Please check the spelling.', 'error'); return; } const employee = workingEmployees[employeeIndex]; const originalManager = employee.manager; const offEmployee = { name: employee.name, manager: originalManager, status: status, type: 'manual', timeAdded: new Date().toLocaleString(), originalManager: originalManager }; offEmployees.push(offEmployee); workingEmployees.splice(employeeIndex, 1); manuallyAdded.set(employee.name, new Date()); nameInput.value = ''; updateDisplay(); forceTableUpdate(); showMessage(`${employee.name} has been marked as "${status}" and kept with ${originalManager}`, 'success'); switchTab('off'); }
