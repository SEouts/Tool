// Check if page loaded properly window.addEventListener('DOMContentLoaded', function() { console.log('Page loaded'); console.log('XLSX library available:', typeof XLSX !== 'undefined'); if (typeof XLSX === 'undefined') { console.error('XLSX library not loaded'); alert('Excel processing library failed to load. Please refresh the page.'); } else { console.log('XLSX library loaded successfully'); } }); let workingEmployees = []; let offEmployees = []; let allManagers = new Set(); let manuallyAdded = new Map(); // Track manually added employees // File upload handlers document.getElementById('scheduleFile').addEventListener('change', handleScheduleFile); document.getElementById('timeOffFile').addEventListener('change', handleTimeOffFile); function handleScheduleFile(event) { const file = event.target.files[0]; if (!file) return; console.log('Processing schedule file:', file.name); // Check if XLSX is loaded if (typeof XLSX === 'undefined') { alert('Excel processing library not loaded. Please refresh the page and try again.'); return; } const reader = new FileReader(); reader.onload = function(e) { try { console.log('Reading schedule file data...'); const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); console.log('Schedule workbook sheets:', workbook.SheetNames); const worksheet = workbook.Sheets[workbook.SheetNames[0]]; const jsonData = XLSX.utils.sheet_to_json(worksheet, { header: 1 }); console.log('Schedule data rows:', jsonData.length); // Process schedule data processScheduleData(jsonData); updateDisplay(); console.log('Schedule processing complete'); } catch (error) { console.error('Schedule file error:', error); alert('Error reading schedule file: ' + error.message); } }; reader.readAsArrayBuffer(file); } function handleTimeOffFile(event) { const file = event.target.files[0]; if (!file) return; console.log('Processing time off file:', file.name); // Check if XLSX is loaded if (typeof XLSX === 'undefined') { alert('Excel processing library not loaded. Please refresh the page and try again.'); return; } const reader = new FileReader(); reader.onload = function(e) { try { console.log('Reading time off file data...'); const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); console.log('Time off workbook sheets:', workbook.SheetNames); const worksheet = workbook.Sheets[workbook.SheetNames[0]]; const jsonData = XLSX.utils.sheet_to_json(worksheet, { header: 1 }); console.log('Time off data rows:', jsonData.length); // Process time off data processTimeOffData(jsonData); updateDisplay(); console.log('Time off processing complete'); } catch (error) { console.error('Time off file error:', error); alert('Error reading time off file: ' + error.message); } }; reader.readAsArrayBuffer(file); } function processScheduleData(data) { console.log('Processing schedule data, rows:', data.length); workingEmployees = []; // Skip header row, process data rows for (let i = 1; i < data.length; i++) { const row = data[i]; if (!row || !row[0]) continue; // Skip empty rows const employee = { name: String(row[0]).trim(), // Worker manager: String(row[1] || '').trim(), // Supervisory Organization schedule: String(row[2] || '').trim(), // Published Schedule type: 'scheduled' }; if (employee.name) { workingEmployees.push(employee); if (employee.manager) { allManagers.add(employee.manager); } } } console.log('Processed working employees:', workingEmployees.length); } function processTimeOffData(data) { console.log('Processing time off data, rows:', data.length); // Skip header row, process data rows for (let i = 1; i < data.length; i++) { const row = data[i]; if (!row || !row[2]) continue; // Skip empty rows (Worker column) const timeOffHours = String(row[0] || ''); // Time Off Entry const isPartialDay = timeOffHours.includes('4 Hours'); const employee = { name: String(row[2]).trim(), // Worker manager: String(row[1] || '').trim(), // Supervisory Organization status: isPartialDay ? 'Partial Day (4 Hours)' : 'Full Day Off', type: isPartialDay ? 'partial' : 'scheduled', timeAdded: 'Scheduled' }; if (employee.name) { offEmployees.push(employee); if (employee.manager) { allManagers.add(employee.manager); } // Remove from working list if they exist there workingEmployees = workingEmployees.filter(emp => emp.name !== employee.name); } } console.log('Processed off employees:', offEmployees.length); console.log('Remaining working employees:', workingEmployees.length); } function switchTab(tabName) { // Update tab buttons document.querySelectorAll('.tab').forEach(tab => tab.classList.remove('active')); document.querySelector(`[onclick="switchTab('${tabName}')"]`).classList.add('active'); // Update tab content document.querySelectorAll('.tab-content').forEach(content => content.classList.remove('active')); document.getElementById(tabName + 'Tab').classList.add('active'); } function markEmployeeWithStatus(status) { const nameInput = document.getElementById('manualEmployeeName'); const employeeName = nameInput.value.trim(); if (!employeeName) { alert('Please enter an employee name'); return; } // Find employee in working list const employeeIndex = workingEmployees.findIndex(emp => emp.name.toLowerCase().includes(employeeName.toLowerCase()) ); if (employeeIndex === -1) { alert('Employee not found in working list. Please check the spelling or ensure they are scheduled to work today.'); return; } const employee = workingEmployees[employeeIndex]; // Move to off list with specific status const offEmployee = { name: employee.name, manager: employee.manager, status: status, type: 'manual', timeAdded: new Date().toLocaleString() }; offEmployees.push(offEmployee); workingEmployees.splice(employeeIndex, 1); manuallyAdded.set(employee.name, new Date()); nameInput.value = ''; updateDisplay(); // Show success message and switch to off tab alert(`${employee.name} has been marked as "${status}" and moved to the Off Today tab.`); switchTab('off'); } const searchInput = document.getElementById('employeeSearch'); const employeeName = searchInput.value.trim(); if (!employeeName) { alert('Please enter an employee name'); return; } // Find employee in working list const employeeIndex = workingEmployees.findIndex(emp => emp.name.toLowerCase().includes(employeeName.toLowerCase()) ); if (employeeIndex === -1) { alert('Employee not found in working list'); return; } const employee = workingEmployees[employeeIndex]; function markEmployeeOff() { const searchInput = document.getElementById('employeeSearch'); const employeeName = searchInput.value.trim(); if (!employeeName) { alert('Please enter an employee name'); return; } // Find employee in working list const employeeIndex = workingEmployees.findIndex(emp => emp.name.toLowerCase().includes(employeeName.toLowerCase()) ); if (employeeIndex === -1) { alert('Employee not found in working list'); return; } const employee = workingEmployees[employeeIndex]; // Move to off list with generic status const offEmployee = { name: employee.name, manager: employee.manager, status: 'Off Today', type: 'manual', timeAdded: new Date().toLocaleString() }; offEmployees.push(offEmployee); workingEmployees.splice(employeeIndex, 1); manuallyAdded.set(employee.name, new Date()); searchInput.value = ''; updateDisplay(); } function removeFromOff(employeeName) { offEmployees = offEmployees.filter(emp => emp.name !== employeeName); manuallyAdded.delete(employeeName); updateDisplay(); } function restoreToWorking(employeeName) { const offEmployeeIndex = offEmployees.findIndex(emp => emp.name === employeeName); if (offEmployeeIndex === -1) return; const employee = offEmployees[offEmployeeIndex]; // Add back to working (you might want to get original schedule info) const workingEmployee = { name: employee.name, manager: employee.manager, schedule: 'Restored - Check original schedule', type: 'restored' }; workingEmployees.push(workingEmployee); offEmployees.splice(offEmployeeIndex, 1); manuallyAdded.delete(employeeName); updateDisplay(); } function filterByManager(tabType) { const selectedManager = document.getElementById(tabType + 'ManagerFilter').value; updateTable(tabType, selectedManager); } function updateDisplay() { updateCounts(); updateManagerFilters(); updateTable('working'); updateTable('off'); updateSummaries(); } function updateCounts() { document.getElementById('workingCount').textContent = workingEmployees.length; document.getElementById('offCount').textContent = offEmployees.length; } function updateManagerFilters() { const workingFilter = document.getElementById('workingManagerFilter'); const offFilter = document.getElementById('offManagerFilter'); // Clear existing options except "All Managers" workingFilter.innerHTML = ''; offFilter.innerHTML = ''; // Add manager options Array.from(allManagers).sort().forEach(manager => { if (manager) { workingFilter.innerHTML += ``; offFilter.innerHTML += ``; } }); } function updateTable(tabType, managerFilter = '') { const tableBody = document.getElementById(tabType + 'TableBody'); const employees = tabType === 'working' ? workingEmployees : offEmployees; let filteredEmployees = employees; if (managerFilter) { filteredEmployees = employees.filter(emp => emp.manager === managerFilter); } if (filteredEmployees.length === 0) { tableBody.innerHTML = `No employees found`; return; } // Sort by manager name filteredEmployees.sort((a, b) => (a.manager || '').localeCompare(b.manager || '')); let html = ''; filteredEmployees.forEach(employee => { if (tabType === 'working') { html += ` ${employee.name} ${employee.manager || 'N/A'} ${employee.schedule || 'N/A'} `; } else { const statusClass = employee.type === 'partial' ? 'status-partial' : employee.type === 'manual' ? 'status-manual' : 'status-scheduled'; html += ` ${employee.name} ${employee.manager || 'N/A'} ${employee.status} ${employee.timeAdded} ${employee.type === 'manual' ? `` : '' } `; } }); tableBody.innerHTML = html; } function markSpecificEmployeeOff(employeeName) { document.getElementById('employeeSearch').value = employeeName; markEmployeeOff(); } function updateSummaries() { // Working summary const workingByManager = {}; workingEmployees.forEach(emp => { const manager = emp.manager || 'No Manager'; workingByManager[manager] = (workingByManager[manager] || 0) + 1; }); let workingSummaryHtml = ''; Object.entries(workingByManager).forEach(([manager, count]) => { workingSummaryHtml += `

${count}

${manager}

`; }); document.getElementById('workingSummary').innerHTML = workingSummaryHtml; // Off summary const offByManager = {}; offEmployees.forEach(emp => { const manager = emp.manager || 'No Manager'; offByManager[manager] = (offByManager[manager] || 0) + 1; }); let offSummaryHtml = ''; Object.entries(offByManager).forEach(([manager, count]) => { offSummaryHtml += `

${count}

${manager}

`; }); document.getElementById('offSummary').innerHTML = offSummaryHtml; } function exportData() { const wb = XLSX.utils.book_new(); // Working employees sheet const workingData = [['Employee', 'Manager', 'Schedule', 'Status']]; workingEmployees.forEach(emp => { workingData.push([emp.name, emp.manager, emp.schedule, 'Working']); }); const workingWs = XLSX.utils.aoa_to_sheet(workingData); XLSX.utils.book_append_sheet(wb, workingWs, 'Working Today'); // Off employees sheet const offData = [['Employee', 'Manager', 'Status', 'Time Added', 'Type']]; offEmployees.forEach(emp => { offData.push([emp.name, emp.manager, emp.status, emp.timeAdded, emp.type]); }); const offWs = XLSX.utils.aoa_to_sheet(offData); XLSX.utils.book_append_sheet(wb, offWs, 'Off Today'); // Export file const fileName = `Employee_Availability_${new Date().toISOString().split('T')[0]}.xlsx`; XLSX.writeFile(wb, fileName); } // Enable search on Enter key document.getElementById('employeeSearch').addEventListener('keypress', function(e) { if (e.key === 'Enter') { markEmployeeOff(); } }); // Enable manual input on Enter key document.addEventListener('keypress', function(e) { if (e.key === 'Enter' && e.target.id === 'manualEmployeeName') { // Default to "Off Today" when pressing Enter markEmployeeWithStatus('Off Today'); } });
